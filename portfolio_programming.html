                    package WarthonAlan_lab2;
                    import java.util.Scanner;
                    
                    public class WhatsUrName {
                        public static void main(String[] args) {
                            Scanner input = new Scanner(System.in);
                            String name;
                            int hours;
                            double fee, totalTuition;
                            
                            System.out.println("What is your name?");
                            name = input.nextLine();
                    
                            System.out.println("How many hours of credit will you take this semester?");
                            hours = input.nextInt();
                            
                            System.out.println("How much is your fee per credit hour?");
                            fee = input.nextDouble();
                            
                            totalTuition = fee * hours;
                            
                            System.out.println("******* r e p o r t *******");
                            System.out.println("Your name is " + name + ".");
                            System.out.println("You will take " + hours + " hours of credit this semester.");
                            System.out.printf("Your total fee of tuition will be $%,.2f this semester.\n", totalTuition);
                        }
                    }
